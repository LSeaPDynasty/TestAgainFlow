**Service层**: `backend/app/services/quick_create_service.py`

```python
"""
快速创建服务 - 智能推断和创建资源
"""
class QuickCreateService:
    """快速创建服务 - 自动处理复杂逻辑"""

    def __init__(self, db: Session):
        self.db = db
        self.ai_service = AIService()  # 使用AI辅助推断

    async def create_from_simple_format(
        self,
        request: QuickTestcaseCreate
    ) -> Dict[str, Any]:
        """
        从简化格式创建完整用例

        智能推断：
        1. 从步骤描述推断Screen
        2. 从target推断Element
        3. 从action推断Step类型
        4. 自动创建Flow
        5. 组装Testcase
        """
        result = {
            "created": {
                "screens": [],
                "elements": [],
                "steps": [],
                "flows": [],
                "testcases": []
            },
            "warnings": []
        }

        # 1. 智能推断Screen（根据步骤序列）
        screens = await self._infer_screens_from_steps(request.steps)

        # 2. 智能推断Element（从target描述）
        elements = await self._infer_elements_from_steps(
            request.steps,
            screens,
            device_serial=request.device_serial
        )

        # 3. 创建Steps
        steps = await self._create_steps_from_simple(
            request.steps,
            elements
        )

        # 4. 创建Flow（自动命名为 {用例名}_flow）
        flow = await self._create_auto_flow(
            f"{request.name}_flow",
            steps
        )

        # 5. 创建Testcase
        testcase = await self._create_testcase(
            request.name,
            request.description,
            flow_id=flow.id
        )

        result["created"]["testcases"].append(testcase.name)
        return result

    async def _infer_screens_from_steps(
        self,
        steps: List[SimpleStep]
    ) -> List[Screen]:
        """
        从步骤序列智能推断Screen

        策略：
        1. 分析步骤的上下文
        2. 识别界面切换点
        3. 创建Screen对象
        """
        inferred_screens = []
        current_screen = None

        for step in steps:
            # 识别界面切换关键词
            if "登录" in step.target or "登录" in step.name:
                screen_name = "登录页"
            elif "首页" in step.target or "主页" in step.target:
                screen_name = "首页"
            elif "设置" in step.target:
                screen_name = "设置页"
            else:
                screen_name = "通用界面"

            # 检查是否已存在
            if not any(s.name == screen_name for s in inferred_screens):
                screen = Screen(name=screen_name)
                self.db.add(screen)
                self.db.flush()
                inferred_screens.append(screen)

        return inferred_screens

    async def _infer_elements_from_steps(
        self,
        steps: List[SimpleStep],
        screens: List[Screen],
        device_serial: Optional[str] = None
    ) -> List[Element]:
        """
        从步骤推断Element

        策略：
        1. 使用AI理解target描述
        2. 如果有设备，从DOM查找
        3. 生成多种定位符
        """
        elements = []

        for step in steps:
            if step.action in ["click", "input", "assert"]:
                # 使用AI推断元素特征
                element_inference = await self.ai_service.infer_element(
                    description=step.target,
                    context=step.name
                )

                # 检查是否已存在相似元素
                existing = await self._find_similar_element(
                    element_inference["name"]
                )

                if existing:
                    elements.append(existing)
                else:
                    # 创建新元素
                    element = Element(
                        name=element_inference["name"],
                        description=step.target,
                        screen_id=screens[0].id,  # 简化：使用第一个screen
                        locators=element_inference["locators"]
                    )
                    self.db.add(element)
                    self.db.flush()
                    elements.append(element)

        return elements

    async def _create_steps_from_simple(
        self,
        simple_steps: List[SimpleStep],
        elements: List[Element]
    ) -> List[Step]:
        """将SimpleStep转换为完整Step"""
        steps = []
        element_map = {e.description: e for e in elements}

        for idx, simple_step in enumerate(simple_steps):
            # 映射action类型
            action_type_map = {
                "click": "click",
                "input": "input",
                "assert": "assert_text",
                "wait": "wait_time"
            }

            step = Step(
                name=simple_step.name,
                description=simple_step.name,
                action_type=action_type_map.get(
                    simple_step.action,
                    simple_step.action
                ),
                element_id=element_map.get(
                    simple_step.target
                , Element()).id if simple_step.target in element_map else None,
                action_value=simple_step.value,
                wait_after_ms=simple_step.wait_ms,
                screen_id=elements[0].screen_id if elements else 1
            )

            self.db.add(step)
            self.db.flush()
            steps.append(step)

        return steps

    async def _create_auto_flow(
        self,
        flow_name: str,
        steps: List[Step]
    ) -> Flow:
        """自动创建Flow"""
        flow = Flow(name=flow_name, description="自动生成的流程")
        self.db.add(flow)
        self.db.flush()

        # 创建FlowStep关联
        for idx, step in enumerate(steps):
            flow_step = FlowStep(
                flow_id=flow.id,
                step_id=step.id,
                order_index=idx + 1
            )
            self.db.add(flow_step)

        self.db.commit()
        return flow

    async def _create_testcase(
        self,
        name: str,
        description: Optional[str],
        flow_id: int
    ) -> Testcase:
        """创建Testcase"""
        testcase = Testcase(
            name=name,
            description=description,
            priority="P1"
        )
        self.db.add(testcase)
        self.db.flush()

        # 关联Main Flow
        testcase_flow = TestcaseFlow(
            testcase_id=testcase.id,
            flow_id=flow_id,
            flow_role="main",
            order_index=1
        )
        self.db.add(testcase_flow)

        self.db.commit()
        return testcase
```

```typescript
import React, { useState } from 'react';
import {
  Steps,
  Form,
  Input,
  Button,
  Card,
  Space,

interface SimpleStep {
  name: string;
  action: string;
  target?: string;
  value?: string;
  wait_ms: number;
}

export const QuickCreateWizard: React.FC = () => {
  const [currentStep, setCurrentStep] = useState(0);
  const [form] = Form.useForm();
  const [steps, setSteps] = useState<SimpleStep[]>([]);
  const [creating, setCreating] = useState(false);

  const actionOptions = [
    { value: 'click', label: '点击' },
    { value: 'input', label: '输入' },
    { value: 'assert', label: '断言' },
    { value: 'wait', label: '等待' },
  ];

  const addStep = () => {
    setSteps([
      ...steps,
      {
        name: `步骤${steps.length + 1}`,
        action: 'click',
        wait_ms: 300
      }
    ]);
  };

  const removeStep = (index: number) => {
    setSteps(steps.filter((_, i) => i !== index));
  };

  const updateStep = (index: number, field: string, value: any) => {
    const newSteps = [...steps];
    newSteps[index] = { ...newSteps[index], [field]: value };
    setSteps(newSteps);
  };

  const handleCreate = async () => {
    try {
      const values = await form.validateFields();

      setCreating(true);
      const response = await quickCreateService.createTestcase({
        name: values.name,
        description: values.description,
        steps: steps
      });

      message.success(`用例 "${values.name}" 创建成功！`);
      setCurrentStep(2); // 跳到完成页

    } catch (error: any) {
      message.error(error.response?.data?.message || '创建失败');
    } finally {
      setCreating(false);
    }
  };

  return (
    <Card title="快速创建测试用例">
      <Steps current={currentStep}>
        <Steps.Step title="基本信息" />
        <Steps.Step title="添加步骤" />
        <Steps.Step title="完成" />
      </Steps>

      {currentStep === 0 && (
        <Form form={form} layout="vertical">
          <Form.Item
            name="name"
            label="用例名称"
            rules={[{ required: true, message: '请输入用例名称' }]}
          >
            <Input placeholder="例如：用户登录测试" />
          </Form.Item>

          <Form.Item
            name="description"
            label="用例描述"
          >
            <Input.TextArea
              placeholder="简要描述测试目的"
              rows={3}
            />
          </Form.Item>

          <Button
            type="primary"
            onClick={() => setCurrentStep(1)}
          >
            下一步：添加步骤
          </Button>
        </Form>
      )}

      {currentStep === 1 && (
        <Space direction="vertical" style={{ width: '100%' }}>
          <div>
            <Button
              type="dashed"
              onClick={addStep}
              icon={<PlusOutlined />}
              block
            >
              添加步骤
            </Button>
          </div>

          <List
            dataSource={steps}
            renderItem={(step, index) => (
              <List.Item
                actions={[
                  <Button
                    danger
                    size="small"
                    icon={<DeleteOutlined />}
                    onClick={() => removeStep(index)}
                  >
                    删除
                  </Button>
                ]}
              >
                <Card size="small" style={{ width: '100%' }}>
                  <Row gutter={16}>
                    <Col span={6}>
                      <Input
                        placeholder="步骤名称"
                        value={step.name}
                        onChange={(e) =>
                          updateStep(index, 'name', e.target.value)
                        }
                      />
                    </Col>

                    <Col span={4}>
                      <Select
                        value={step.action}
                        onChange={(value) =>
                          updateStep(index, 'action', value)
                        }
                        options={actionOptions}
                        style={{ width: '100%' }}
                      />
                    </Col>

                    <Col span={6}>
                      <Input
                        placeholder="目标元素"
                        value={step.target}
                        onChange={(e) =>
                          updateStep(index, 'target', e.target.value)
                        }
                        disabled={step.action === 'wait'}
                      />
                    </Col>

                    <Col span={4}>
                      <Input
                        placeholder="输入值"
                        value={step.value}
                        onChange={(e) =>
                          updateStep(index, 'value', e.target.value)
                        }
                        disabled={step.action !== 'input'}
                      />
                    </Col>

                    <Col span={4}>
                      <InputNumber
                        placeholder="等待(ms)"
                        value={step.wait_ms}
                        onChange={(value) =>
                          updateStep(index, 'wait_ms', value || 300)
                        }
                        min={0}
                        max={10000}
                        style={{ width: '100%' }}
                      />
                    </Col>
                  </Row>
                </Card>
              </List.Item>
            )}
          />

          <Space>
            <Button onClick={() => setCurrentStep(0)}>
              上一步
            </Button>
            <Button
              type="primary"
              onClick={handleCreate}
              loading={creating}
              disabled={steps.length === 0}
            >
              创建用例
            </Button>
          </Space>
        </Space>
      )}

      {currentStep === 2 && (
        <div style={{ textAlign: 'center', padding: '40px 0' }}>
          <CheckOutlined
            style={{ fontSize: 64, color: '#52c41a', marginBottom: 16 }}
          />

          <Button type="primary" onClick={() => window.location.href = '/testcases'}>
            查看用例
          </Button>
        </div>
      )}
    </Card>
  );
```

```typescript
/**
 * 流程可视化编辑器
 */
import React, { useCallback, useMemo } from 'react';
import ReactFlow, {
  Node,
  Edge,
  addEdge,
  Connection,
  useNodesState,
  useEdgesState,
  Background,
  Controls,
  MiniMap,
  NodeTypes,
} from 'reactflow';
import 'reactflow/dist/style.css';
import { StepNode } from './nodes/StepNode';
import { FlowNode } from './nodes/FlowNode';
import { StartNode } from './nodes/StartNode';
import { EndNode } from './nodes/EndNode';

const nodeTypes: NodeTypes = {
  step: StepNode,
  flow: FlowNode,
  start: StartNode,
  end: EndNode,
};

interface FlowVisualizerProps {
  flowId: number;
  readonly?: boolean;
  onSave?: (nodes: Node[], edges: Edge[]) => void;
}

export const FlowVisualizer: React.FC<FlowVisualizerProps> = ({
  flowId,
  readonly = false,
  onSave
}) => {
  const [nodes, setNodes, onNodesChange] = useNodesState([]);
  const [edges, setEdges, onEdgesChange] = useEdgesState([]);

  // 加载Flow数据
  React.useEffect(() => {
    loadFlowData();
  }, [flowId]);

  const loadFlowData = async () => {
    // 从API加载Flow数据
    const response = await flowService.getFlowDetail(flowId);
    const flowData = response.data;

    // 转换为React Flow格式
    const flowNodes: Node[] = [];
    const flowEdges: Edge[] = [];

    // 添加开始节点
    flowNodes.push({
      id: 'start',
      type: 'start',
      position: { x: 100, y: 100 },
      data: { label: '开始' }
    });

    // 添加步骤节点
    flowData.flow_steps.forEach((fs: any, index: number) => {
      const nodeId = fs.step_id ? `step-${fs.step_id}` : `flow-${fs.sub_flow_id}`;
      const nodeType = fs.step_id ? 'step' : 'flow';

      flowNodes.push({
        id: nodeId,
        type: nodeType,
        position: {
          x: 100,
          y: 150 + index * 150
        },
        data: {
          ...fs,
          label: fs.step?.name || fs.sub_flow?.name
        }
      });

      // 添加连接边
      const prevNodeId = index === 0 ? 'start' : flowNodes[index].id;
      flowEdges.push({
        id: `edge-${prevNodeId}-${nodeId}`,
        source: prevNodeId,
        target: nodeId,
        type: 'smoothstep',
        animated: true
      });
    });

    // 添加结束节点
    flowNodes.push({
      id: 'end',
      type: 'end',
      position: {
        x: 100,
        y: 150 + flowData.flow_steps.length * 150
      },
      data: { label: '结束' }
    });

    setNodes(flowNodes);
    setEdges(flowEdges);
  };

  const onConnect = useCallback(
    (params: Connection) => setEdges((eds) => addEdge(params, eds)),
    [setEdges]
  );

  const handleSave = () => {
    if (onSave) {
      // 将React Flow格式转换回数据库格式
      const flowSteps = nodes
        .filter((n) => n.type === 'step' || n.type === 'flow')
        .map((n, index) => ({
          step_id: n.type === 'step' ? parseInt(n.id.split('-')[1]) : null,
          sub_flow_id: n.type === 'flow' ? parseInt(n.id.split('-')[1]) : null,
          order_index: index + 1
        }));

      onSave(nodes, edges);
    }
  };

  return (
    <div style={{ height: '600px', width: '100%' }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={readonly ? undefined : onNodesChange}
        onEdgesChange={readonly ? undefined : onEdgesChange}
        onConnect={readonly ? undefined : onConnect}
        nodeTypes={nodeTypes}
        fitView
        attributionPosition="bottom-left"
      >
        <Background />
        <Controls />
        <MiniMap />
      </ReactFlow>

      {!readonly && (
        <div style={{ position: 'absolute', top: 10, right: 10 }}>
          <Button type="primary" onClick={handleSave}>
            保存流程
          </Button>
        </div>
      )}
    </div>
  );
};
```

```python
class TemplateService:
    """模板管理服务"""

    def __init__(self, db: Session):
        self.db = db

    async def create_from_template(
        self,
        template_id: int,
        testcase_name: str,
        variables: Dict[str, Any]
    ) -> Testcase:
        """
        从模板创建用例

        流程：
        1. 加载模板
        2. 替换变量
        3. 创建资源
        4. 组装用例
        """
        # 1. 加载模板
        template = self.db.query(TestTemplate).filter_by(
            id=template_id
        ).first()

        if not template:
            raise ValueError("Template not found")

        # 2. 替换变量
        template_data = self._replace_variables(
            template.template_data,
            {**template.variables, **variables}
        )

        # 3. 创建Flow
        flows = {}
        for role in ['setup', 'main', 'teardown']:
            for flow_def in template_data.get(f"{role}_flows", []):
                flow = await self._create_flow_from_def(flow_def)
                flows[f"{role}_{flow.name}"] = flow

        # 4. 创建Testcase
        testcase = Testcase(
            name=testcase_name,
            description=f"基于模板 {template.name} 创建",
            priority="P1"
        )
        self.db.add(testcase)
        self.db.flush()

        # 5. 关联Flows
        for role in ['setup', 'main', 'teardown']:
            for flow_name in template_data.get(f"{role}_flows", []):
                flow = flows[f"{role}_{flow_name}"]
                testcase_flow = TestcaseFlow(
                    testcase_id=testcase.id,
                    flow_id=flow.id,
                    flow_role=role
                )
                self.db.add(testcase_flow)

        self.db.commit()
        return testcase

    def _replace_variables(
        self,
        template_data: Dict,
        variables: Dict[str, Any]
    ) -> Dict:
        """递归替换模板中的变量"""
        import json

        json_str = json.dumps(template_data)

        for key, value in variables.items():
            placeholder = f"{{{{{key}}}}}"
            json_str = json_str.replace(placeholder, str(value))

        return json.loads(json_str)
```

## 4. 依赖分析实施方案

### 目标
在删除/修改资源前，分析影响范围

### 方案：递归依赖分析

#### Backend 依赖分析服务

**Service**: `backend/app/services/impact_analyzer.py`

```python
"""
影响分析服务
"""
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from typing import Dict, List, Set

class ImpactAnalyzer:
    """影响分析服务"""

    def __init__(self, db: Session):
        self.db = db

    def analyze_element_impact(self, element_id: int) -> Dict[str, any]:
        """
        分析元素的影响范围

        返回所有依赖此元素的资源
        """
        impact = {
            "element": None,
            "direct_impact": {
                "steps": [],
                "count": 0
            },
            "indirect_impact": {
                "flows": [],
                "testcases": [],
                "suites": []
            },
            "risk_level": "LOW"
        }

        # 1. 获取元素信息
        element = self.db.query(Element).filter_by(id=element_id).first()
        if not element:
            return impact

        impact["element"] = {
            "id": element.id,
            "name": element.name,
            "screen": element.screen.name
        }

        # 2. 查找直接依赖的Steps
        dependent_steps = self.db.query(Step).filter(
            Step.element_id == element_id
        ).all()

        impact["direct_impact"]["steps"] = [
            {"id": s.id, "name": s.name}
            for s in dependent_steps
        ]
        impact["direct_impact"]["count"] = len(dependent_steps)

        # 3. 递归查找间接影响
        visited_flows: Set[int] = set()
        visited_testcases: Set[int] = set()
        visited_suites: Set[int] = set()

        for step in dependent_steps:
            # 查找使用此Step的Flows
            self._find_flows_using_step(
                step.id,
                visited_flows,
                visited_testcases,
                visited_suites
            )

        impact["indirect_impact"]["flows"] = [
            {"id": fid, "name": self._get_flow_name(fid)}
            for fid in visited_flows
        ]
        impact["indirect_impact"]["testcases"] = [
            {"id": tid, "name": self._get_testcase_name(tid)}
            for tid in visited_testcases
        ]
        impact["indirect_impact"]["suites"] = [
            {"id": sid, "name": self._get_suite_name(sid)}
            for sid in visited_suites
        ]

        # 4. 计算风险等级
        total_impact = (
            len(dependent_steps) +
            len(visited_flows) +
            len(visited_testcases) +
            len(visited_suites)
        )

        if total_impact > 50:
            impact["risk_level"] = "CRITICAL"
        elif total_impact > 20:
            impact["risk_level"] = "HIGH"
        elif total_impact > 5:
            impact["risk_level"] = "MEDIUM"
        else:
            impact["risk_level"] = "LOW"

        return impact

    def _find_flows_using_step(
        self,
        step_id: int,
        visited_flows: Set[int],
        visited_testcases: Set[int],
        visited_suites: Set[int]
    ):
        """递归查找使用Step的Flow及其上级资源"""

        # 查找包含此Step的FlowSteps
        flow_steps = self.db.query(FlowStep).filter(
            FlowStep.step_id == step_id
        ).all()

        for fs in flow_steps:
            flow_id = fs.flow_id

            if flow_id not in visited_flows:
                visited_flows.add(flow_id)

                # 查找使用此Flow的Testcases
                self._find_testcases_using_flow(
                    flow_id,
                    visited_testcases,
                    visited_suites
                )

    def _find_testcases_using_flow(
        self,
        flow_id: int,
        visited_testcases: Set[int],
        visited_suites: Set[int]
    ):
        """递归查找使用Flow的Testcase及其上级资源"""

        # 查找包含此Flow的TestcaseFlows
        testcase_flows = self.db.query(TestcaseFlow).filter(
            TestcaseFlow.flow_id == flow_id
        ).all()

        for tf in testcase_flows:
            testcase_id = tf.testcase_id

            if testcase_id not in visited_testcases:
                visited_testcases.add(testcase_id)

                # 查找包含此Testcase的Suites
                self._find_suites_using_testcase(
                    testcase_id,
                    visited_suites
                )

    def _find_suites_using_testcase(
        self,
        testcase_id: int,
        visited_suites: Set[int]
    ):
        """查找包含Testcase的Suites"""

        suite_testcases = self.db.query(SuiteTestcase).filter(
            SuiteTestcase.testcase_id == testcase_id
        ).all()

        for st in suite_testcases:
            visited_suites.add(st.suite_id)

    def _get_flow_name(self, flow_id: int) -> str:
        flow = self.db.query(Flow).filter_by(id=flow_id).first()
        return flow.name if flow else "Unknown"

    def _get_testcase_name(self, testcase_id: int) -> str:
        testcase = self.db.query(Testcase).filter_by(id=testcase_id).first()
        return testcase.name if testcase else "Unknown"

    def _get_suite_name(self, suite_id: int) -> str:
        suite = self.db.query(Suite).filter_by(id=suite_id).first()
        return suite.name if suite else "Unknown"
```

### 方案：DFS检测算法

#### Backend 循环依赖检测

**Service**: `backend/app/services/cycle_detector.py`

```python
"""
循环依赖检测服务
"""
from sqlalchemy.orm import Session
from typing import Dict, List, Set
from collections import defaultdict

class CycleDetector:
    """循环依赖检测器"""

    def __init__(self, db: Session):
        self.db = db

    def detect_flow_cycle(
        self,
        parent_flow_id: int,
        child_flow_id: int
    ) -> Dict[str, any]:
        """
        检测添加子流程是否会形成循环依赖

        返回:
        {
            "has_cycle": bool,
            "cycle_path": List[str]  # 如果有循环，返回循环路径
        }
        """
        # 1. 构建依赖图
        dependency_graph = self._build_flow_dependency_graph()

        # 2. 添加新的边
        dependency_graph[parent_flow_id].add(child_flow_id)

        # 3. DFS检测循环
        visited: Set[int] = set()
        rec_stack: Set[int] = set()
        path: List[int] = []

        has_cycle, cycle_path = self._dfs_cycle_detection(
            parent_flow_id,
            dependency_graph,
            visited,
            rec_stack,
            path
        )

        if has_cycle:
            # 转换ID为名称
            cycle_names = [
                self._get_flow_name(fid)
                for fid in cycle_path
            ]
            return {
                "has_cycle": True,
                "cycle_path": cycle_names
            }

        return {
            "has_cycle": False,
            "cycle_path": []
        }

    def _build_flow_dependency_graph(self) -> Dict[int, Set[int]]:
        """
        构建Flow依赖图

        返回: {flow_id: Set[依赖的flow_ids]}
        """
        graph = defaultdict(set)

        # 查询所有FlowStep
        flow_steps = self.db.query(FlowStep).filter(
            FlowStep.sub_flow_id.isnot(None)
        ).all()

        for fs in flow_steps:
            # parent_flow 依赖于 child_flow
            graph[fs.flow_id].add(fs.sub_flow_id)

        return graph

    def _dfs_cycle_detection(
        self,
        node: int,
        graph: Dict[int, Set[int]],
        visited: Set[int],
        rec_stack: Set[int],
        path: List[int]
    ) -> tuple[bool, List[int]]:
        """
        DFS检测循环依赖

        返回: (是否有循环, 循环路径)
        """
        visited.add(node)
        rec_stack.add(node)
        path.append(node)

        # 遍历所有邻居
        for neighbor in graph[node]:
            if neighbor not in visited:
                has_cycle, cycle_path = self._dfs_cycle_detection(
                    neighbor,
                    graph,
                    visited,
                    rec_stack,
                    path
                )
                if has_cycle:
                    return True, cycle_path
            elif neighbor in rec_stack:
                # 找到循环！
                cycle_start = path.index(neighbor)
                cycle_path = path[cycle_start:] + [neighbor]
                return True, cycle_path

        path.pop()
        rec_stack.remove(node)
        return False, []

    def _get_flow_name(self, flow_id: int) -> str:
        flow = self.db.query(Flow).filter_by(id=flow_id).first()
        return flow.name if flow else f"Flow-{flow_id}"
```

```python
"""
查询缓存装饰器
"""
from functools import wraps
import hashlib
import json
from typing import Any

def cache_result(ttl_seconds: int = 300):
    """
    缓存查询结果

    使用示例:
    @cache_result(ttl_seconds=300)
    def get_flow(self, flow_id: int):
        return self.db.query(Flow).filter_by(id=flow_id).first()
    """
    def decorator(func):
        _cache = {}

        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = hashlib.md5(
                f"{func.__name__}{args}{kwargs}".encode()
            ).hexdigest()

            # 检查缓存
            if cache_key in _cache:
                result, timestamp = _cache[cache_key]
                import time
                if time.time() - timestamp < ttl_seconds:
                    return result

            # 执行查询
            result = func(*args, **kwargs)

            # 存入缓存
            import time
            _cache[cache_key] = (result, time.time())

            return result

        return wrapper
    return decorator
```

```python
class FailureDiagnostic:
    """失败诊断分析器"""

    def __init__(self, db: Session):
        self.db = db

    def analyze_failure(
        self,
        run_id: int
    ) -> Dict[str, any]:
        """
        分析执行失败的原因

        返回诊断报告
        """
        # 1. 获取执行日志
        run = self.db.query(Run).filter_by(id=run_id).first()
        logs = json.loads(run.logs or "[]")

        # 2. 找到第一个失败的步骤
        failed_step = None
        for log in logs:
            if log.get("event") == "step_failure":
                failed_step = log
                break

        if not failed_step:
            return {
                "diagnosis": "未找到失败的步骤",
                "suggestions": []
            }

        # 3. 分析失败原因
        error_type = failed_step.get("error_type")
        error_message = failed_step.get("error_message")
        context = failed_step.get("context", {})

        diagnosis = {
            "failed_step": failed_step.get("step_name"),
            "error_type": error_type,
            "error_message": error_message,
            "root_cause": self._identify_root_cause(error_type, error_message, context),
            "suggestions": self._generate_suggestions(error_type, context),
            "screenshot": failed_step.get("screenshot"),
            "context": context
        }

        return diagnosis

    def _identify_root_cause(
        self,
        error_type: str,
        error_message: str,
        context: Dict
    ) -> str:
        """识别根本原因"""

        if "Element not found" in error_message:
            element_info = context.get("element_info", {})
            return f"元素未找到: {element_info.get('name')}。可能原因：1) 元素定位符错误 2) 页面未加载完成 3) 元素属性变化"

        elif "Timeout" in error_type:
            return "操作超时。可能原因：1) 网络延迟 2) 服务器响应慢 3) 设备性能问题"

        elif "Assertion" in error_type:
            return "断言失败。实际值与期望值不符"

        else:
            return "未知错误，需要人工分析"

    def _generate_suggestions(
        self,
        error_type: str,
        context: Dict
    ) -> List[str]:
        """生成修复建议"""

        suggestions = []

        if "Element not found" in error_type:
            suggestions.append("建议：增加等待时间，确保页面加载完成")
            suggestions.append("建议：添加备用定位符")
            suggestions.append("建议：检查元素是否存在（使用DOM查看器）")

            suggestions.append("建议：检查网络连接")
            suggestions.append("建议：优化服务器性能")
```

```python
"""
版本管理服务
"""
class VersionManager:
    """版本管理器"""

    def __init__(self, db: Session):
        self.db = db

    def create_version(
        self,
        element_id: int,
        change_reason: str = ""
    ):
        """
        创建元素版本快照

        在更新元素前调用
        """
        element = self.db.query(Element).filter_by(id=element_id).first()
        if not element:
            return

        # 获取当前版本号
        current_version = self.db.query(func.max(ElementVersion.version)).filter(
            ElementVersion.element_id == element_id
        ).scalar() or 0

        # 创建快照
        snapshot = {
            "name": element.name,
            "screen_id": element.screen_id,
            "description": element.description,
            "locators": [
                {
                    "type": l.type,
                    "value": l.value,
                    "priority": l.priority
                }
                for l in element.locators
            ]
        }

        version = ElementVersion(
            element_id=element_id,
            version=current_version + 1,
            snapshot=snapshot,
            change_reason=change_reason
        )

        self.db.add(version)
        self.db.commit()

    def rollback_to_version(
        self,
        element_id: int,
        target_version: int
    ) -> Element:
        """回滚到指定版本"""
        # 获取目标版本快照
        version = self.db.query(ElementVersion).filter(
            ElementVersion.element_id == element_id,
            ElementVersion.version == target_version
        ).first()

        if not version:
            raise ValueError("Version not found")

        # 恢复快照
        element = self.db.query(Element).filter_by(id=element_id).first()

        # 先创建当前版本的备份
        self.create_version(element_id, "Rollback backup")

        # 恢复数据
        snapshot = version.snapshot
        element.name = snapshot["name"]
        element.screen_id = snapshot["screen_id"]
        element.description = snapshot["description"]

        # 删除旧定位符，创建新定位符
        for locator in element.locators:
            self.db.delete(locator)

        for loc_data in snapshot["locators"]:
            locator = Locator(
                element_id=element.id,
                **loc_data
            )
            self.db.add(locator)

        self.db.commit()
        return element

    def get_version_history(
        self,
        element_id: int
    ) -> List[Dict]:
        """获取版本历史"""
        versions = self.db.query(ElementVersion).filter(
            ElementVersion.element_id == element_id
        ).order_by(ElementVersion.version.desc()).all()

        return [
            {
                "version": v.version,
                "change_reason": v.change_reason,
                "created_at": v.created_at.isoformat(),
                "snapshot": v.snapshot
            }
            for v in versions
        ]
```
